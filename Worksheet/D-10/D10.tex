def conjugate_gradient(A, b, x0, eta, Imax):
    x = x0
    r = b - A@x
    d = r 
    n0 = np.dot(r,r)
    n1 = n0
    res = []
    i = 0
    while n0 > eta, and i < Imax:
        z = A@d
        alpha = n0 / (np.dot(z,d))
        x = x + alpha * d
        r = r - alpha * z
        n1 = np.dot(r,r)
        beta = n1 / n0
        d = r + beta * d
        n0 = n1
        i = i+1
        res = [res, np.sqrt(n1)]
        if np.sqrt(n1) > 1e10:
            break
    return x, i, res
\end{verbatim}
